379  std::string help(size_t wid=30, std::string prev="") const {
380  // Delegate to subcommand if needed
381  if(prev == "")
382  prev = progname;
383  else
384  prev += " " + name;
385 
386  if(subcommand != nullptr)
387  return subcommand->help(wid, prev);
388 
389  std::stringstream out;
390  out << prog_description << std::endl;
391  out << "Usage: " << prev;
392 
393  // Check for options
394  bool npos = false;
395  std::set<std::string> groups;
396  for(const Option_p &opt : options) {
397  if(opt->nonpositional()) {
398  npos = true;
399  groups.insert(opt->get_group());
400  }
401  }
402 
403  if(npos)
404  out << " [OPTIONS]";
405 
406  // Positionals
407  bool pos=false;
408  for(const Option_p &opt : options)
409  if(opt->get_positional()) {
410  out << " " << opt->help_positional();
411  if(opt->has_description())
412  pos=true;
413  }
414 
415  if(subcommands.size() > 0) {
416  if(required_subcommand)
417  out << " SUBCOMMAND";
418  else
419  out << " [SUBCOMMAND]";
420  }
421 
422  out << std::endl << std::endl;
423 
424  // Positional descriptions
425  if(pos) {
426  out << "Positionals:" << std::endl;
427  for(const Option_p &opt : options)
428  if(opt->get_positional() && opt->has_description())
429  detail::format_help(out, opt->help_pname(), opt->get_description(), wid);
430  out << std::endl;
431 
432  }
433 
434 
435  // Options
436  if(npos) {
437  for (const std::string& group : groups) {
438  out << group << ":" << std::endl;
439  for(const Option_p &opt : options) {
440  if(opt->nonpositional() && opt->get_group() == group)
441  detail::format_help(out, opt->help_name(), opt->get_description(), wid);
442 
443  }

446  }

481  Classifer _recognize(std::string current) const {
482  std::string dummy1, dummy2;
483 
484  if(current == "--")
485  return Classifer::POSITIONAL_MARK;
486  for(const App_p &com : subcommands) {
487  if(com->name == current)
488  return Classifer::SUBCOMMAND;
489  }
490  if(detail::split_long(current, dummy1, dummy2))
491  return Classifer::LONG;
492  if(detail::split_short(current, dummy1, dummy2))
493  return Classifer::SHORT;
494  return Classifer::NONE;

499  void _parse(std::vector<std::string> &args) {
500  parsed = true;
501 
502  bool positional_only = false;
503 
504  while(args.size()>0) {
505 
506 
507  Classifer classifer = positional_only ? Classifer::NONE : _recognize(args.back());
508  switch(classifer) {
509  case Classifer::POSITIONAL_MARK:
510  args.pop_back();
511  positional_only = true;
512  break;
513  case Classifer::SUBCOMMAND:
514  _parse_subcommand(args);
515  break;
516  case Classifer::LONG:
517  _parse_long(args);
518  break;
519  case Classifer::SHORT:
520  _parse_short(args);
521  break;
522  case Classifer::NONE:
523  positionals.push_back(args.back());
524  args.pop_back();
525  }
526  }
527 
528  if (help_flag != nullptr && help_flag->count() > 0) {
529  throw CallForHelp();
530  }
531 
532 
533  // Collect positionals
534  for(const Option_p& opt : options) {
535  while (opt->get_positional() && opt->count() < opt->get_expected() && positionals.size() > 0) {
536  opt->get_new();
537  opt->add_result(0, positionals.front());
538  positionals.pop_front();
539  }
540  }
541 
542  // Process an INI file
543  if (ini_setting != nullptr && ini_file != "") {
544  try {
545  std::vector<std::string> values = detail::parse_ini(ini_file);
546  while(values.size() > 0) {
547  _parse_long(values, false);
548  }
549 
550  } catch (const FileError &e) {
551  if(ini_required)
552  throw;
553  }
554  }
555 
556 
557  // Get envname options if not yet passed
558  for(const Option_p& opt : options) {
559  if (opt->count() == 0 && opt->_envname != "") {
560  char *ename = std::getenv(opt->_envname.c_str());
561  if(ename != nullptr) {
562  opt->get_new();
563  opt->add_result(0, std::string(ename));
564  }
565  }
566  }
567 
568  // Process callbacks
569  for(const Option_p& opt : options) {
570  if (opt->count() > 0) {
571  opt->run_callback();
572  }
573  }
574 
575  // Verify required options
576  for(const Option_p& opt : options) {
577  // Required
578  if (opt->get_required()
579  && (opt->count() < opt->get_expected() || opt->count() == 0))
580  throw RequiredError(opt->get_name());
581  // Requires
582  for (const Option* opt_req : opt->_requires)
583  if (opt->count() > 0 && opt_req->count() == 0)
584  throw RequiresError(opt->get_name(), opt_req->get_name());
585  // Excludes
586  for (const Option* opt_ex : opt->_excludes)
587  if (opt->count() > 0 && opt_ex->count() != 0)
588  throw ExcludesError(opt->get_name(), opt_ex->get_name());
589  }
590 
591  if(required_subcommand && subcommand == nullptr)
592  throw RequiredError("Subcommand required");
593 
594  if(positionals.size()>0)
595  throw PositionalError("[" + detail::join(positionals) + "]");
596 
597  pre_callback();
598  run_callback();
599  }

615  void _parse_short(std::vector<std::string> &args) {
616  std::string current = args.back();
617 
618  std::string name;
619  std::string rest;
620  if(!detail::split_short(current, name, rest))
621  throw HorribleError("Short");
622  args.pop_back();
623 
624  auto op_ptr = std::find_if(std::begin(options), std::end(options), [name](const Option_p &opt){return opt->check_sname(name);});
625 
626  if(op_ptr == std::end(options)) {
627  missing_options.push_back("-" + name);
628  return;
629  }
630 
631  // Get a reference to the pointer to make syntax bearable
632  Option_p& op = *op_ptr;
633 
634  int vnum = op->get_new();
635  int num = op->get_expected();
636 
637  if(num == 0)
638  op->add_result(vnum, "");
639  else if(rest!="") {
640  if (num > 0)
641  num--;
642  op->add_result(vnum, rest);
643  rest = "";
644  }
645 
646 
647  if(num == -1) {
648  while(args.size()>0 && _recognize(args.back()) == Classifer::NONE) {
649  op->add_result(vnum, args.back());
650  args.pop_back();
651  }
652  } else while(num>0 && args.size() > 0) {
653  num--;
654  std::string current = args.back();
655  args.pop_back();
656  op->add_result(vnum,current);
657  }
658 
659  if(rest != "") {
660  rest = "-" + rest;
661  args.push_back(rest);
662  }
663  }

666  void _parse_long(std::vector<std::string> &args, bool overwrite=true) {
667  std::string current = args.back();
668 
669  std::string name;
670  std::string value;
671  if(!detail::split_long(current, name, value))
672  throw HorribleError("Long");
673  args.pop_back();
674 
675  auto op_ptr = std::find_if(std::begin(options), std::end(options), [name](const Option_p &v){return v->check_lname(name);});
676 
677  if(op_ptr == std::end(options)) {
678  missing_options.push_back("--" + name);
679  return;
680  }
681 
682  // Get a reference to the pointer to make syntax bearable
683  Option_p& op = *op_ptr;
684 
685 
686  // Stop if not overwriting options (for ini parse)
687  if(!overwrite && op->count() > 0)
688  return;
689 
690  int vnum = op->get_new();
691  int num = op->get_expected();
692 
693 
694  if(value != "") {
695  if(num!=-1) num--;
696  op->add_result(vnum, value);
697  } else if (num == 0) {
698  op->add_result(vnum, "");
699  }
700 
701  if(num == -1) {
702  while(args.size() > 0 && _recognize(args.back()) == Classifer::NONE) {
703  op->add_result(vnum, args.back());
704  args.pop_back();
705  }
706  } else while(num>0 && args.size()>0) {
707  num--;
708  op->add_result(vnum,args.back());
709  args.pop_back();
710  }
711  return;
712  }
