669  std::string config_to_str(bool default_also = false, std::string prefix = "") const {
670  std::stringstream out;
671  for(const Option_p &opt : options_) {
672 
673  // Only process option with a long-name
674  if(!opt->lnames_.empty()) {
675  std::string name = prefix + opt->lnames_[0];
676 
677  // Non-flags
678  if(opt->get_expected() != 0) {
679 
680  // If the option was found on command line
681  if(opt->count() > 0)
682  out << name << "=" << detail::inijoin(opt->results()) << std::endl;
683 
684  // If the option has a default and is requested by optional argument
685  else if(default_also && opt->defaultval_ != "")
686  out << name << "=" << opt->defaultval_ << std::endl;
687  // Flag, one passed
688  } else if(opt->count() == 1) {
689  out << name << "=true" << std::endl;
690 
691  // Flag, multiple passed
692  } else if(opt->count() > 1) {
693  out << name << "=" << opt->count() << std::endl;
694 
695  // Flag, not present
696  } else if(opt->count() == 0 && default_also && opt.get() != get_help_ptr()) {
697  out << name << "=false" << std::endl;
698  }
699  }
700  }
701  for(const App_p &subcom : subcommands_)
702  out << subcom->config_to_str(default_also, prefix + subcom->name_ + ".");
703  return out.str();
704  }

707  std::string help(size_t wid = 30, std::string prev = "") const {
708  // Delegate to subcommand if needed
709  if(prev == "")
710  prev = name_;
711  else
712  prev += " " + name_;
713 
714  auto selected_subcommands = get_subcommands();
715  if(!selected_subcommands.empty())
716  return selected_subcommands.at(0)->help(wid, prev);
717 
718  std::stringstream out;
719  out << description_ << std::endl;
720  out << "Usage: " << prev;
721 
722  // Check for options_
723  bool npos = false;
724  std::set<std::string> groups;
725  for(const Option_p &opt : options_) {
726  if(opt->nonpositional()) {
727  npos = true;
728  groups.insert(opt->get_group());
729  }
730  }
731 
732  if(npos)
733  out << " [OPTIONS]";
734 
735  // Positionals
736  bool pos = false;
737  for(const Option_p &opt : options_)
738  if(opt->get_positional()) {
739  // A hidden positional should still show up in the usage statement
740  // if(detail::to_lower(opt->get_group()) == "hidden")
741  // continue;
742  out << " " << opt->help_positional();
743  if(opt->_has_help_positional())
744  pos = true;
745  }
746 
747  if(!subcommands_.empty()) {
748  if(require_subcommand_ != 0)
749  out << " SUBCOMMAND";
750  else
751  out << " [SUBCOMMAND]";
752  }
753 
754  out << std::endl << std::endl;
755 
756  // Positional descriptions
757  if(pos) {
758  out << "Positionals:" << std::endl;
759  for(const Option_p &opt : options_) {
760  if(detail::to_lower(opt->get_group()) == "hidden")
761  continue;
762  if(opt->_has_help_positional())
763  detail::format_help(out, opt->help_pname(), opt->get_description(), wid);
764  }
765  out << std::endl;
766  }
767 
768  // Options
769  if(npos) {
770  for(const std::string &group : groups) {
771  if(detail::to_lower(group) == "hidden")
772  continue;
773  out << group << ":" << std::endl;
774  for(const Option_p &opt : options_) {
775  if(opt->nonpositional() && opt->get_group() == group)
776  detail::format_help(out, opt->help_name(), opt->get_description(), wid);
777  }
778  out << std::endl;
779  }
780  }
781 
782  // Subcommands
783  if(!subcommands_.empty()) {
784  out << "Subcommands:" << std::endl;
785  for(const App_p &com : subcommands_)
786  detail::format_help(out, com->get_name(), com->description_, wid);
787  }
788  return out.str();
789  }

867  detail::Classifer _recognize(const std::string &current) const {
868  std::string dummy1, dummy2;
869 
870  if(current == "--")
871  return detail::Classifer::POSITIONAL_MARK;
872  if(_valid_subcommand(current))
873  return detail::Classifer::SUBCOMMAND;
874  if(detail::split_long(current, dummy1, dummy2))
875  return detail::Classifer::LONG;
876  if(detail::split_short(current, dummy1, dummy2))

882  void _parse(std::vector<std::string> &args) {
883  parsed_ = true;
884  bool positional_only = false;
885 
886  while(!args.empty()) {
887  _parse_single(args, positional_only);
888  }
889 
890  if(help_ptr_ != nullptr && help_ptr_->count() > 0) {
891  throw CallForHelp();
892  }
893 
894  // Process an INI file
895  if(config_ptr_ != nullptr && config_name_ != "") {
896  try {
897  std::vector<detail::ini_ret_t> values = detail::parse_ini(config_name_);
898  while(!values.empty()) {
899  if(!_parse_ini(values)) {
900  throw ExtrasINIError(values.back().fullname);
901  }
902  }
903  } catch(const FileError &) {
904  if(config_required_)
905  throw;
906  }
907  }

909  // Get envname options if not yet passed
910  for(const Option_p &opt : options_) {
911  if(opt->count() == 0 && opt->envname_ != "") {
912  char *buffer = nullptr;
913  std::string ename_string;
914 
915  #ifdef _MSC_VER
916  // Windows version
917  size_t sz = 0;
918  if(_dupenv_s(&buffer, &sz, opt->envname_.c_str()) == 0 && buffer != nullptr) {
919  ename_string = std::string(buffer);
920  free(buffer);
921  }
922  #else
923  // This also works on Windows, but gives a warning
924  buffer = std::getenv(opt->envname_.c_str());
925  if(buffer != nullptr)
926  ename_string = std::string(buffer);
927  #endif
928 
929  if(!ename_string.empty()) {
930  opt->add_result(ename_string);
931  }
932  }
933  }
934 
935  // Process callbacks
936  for(const Option_p &opt : options_) {
937  if(opt->count() > 0 && !opt->get_callback_run()) {
938  opt->run_callback();
939  }
940  }
941 
942  // Verify required options
943  for(const Option_p &opt : options_) {
944  // Required
945  if(opt->get_required() && (static_cast<int>(opt->count()) < opt->get_expected() || opt->count() == 0))
946  throw RequiredError(opt->get_name());
947  // Requires
948  for(const Option *opt_req : opt->requires_)
949  if(opt->count() > 0 && opt_req->count() == 0)
950  throw RequiresError(opt->get_name(), opt_req->get_name());
951  // Excludes
952  for(const Option *opt_ex : opt->excludes_)
953  if(opt->count() > 0 && opt_ex->count() != 0)
954  throw ExcludesError(opt->get_name(), opt_ex->get_name());
955  }
956 
957  auto selected_subcommands = get_subcommands();
958  if(require_subcommand_ < 0 && selected_subcommands.empty())
959  throw RequiredError("Subcommand required");
960  else if(require_subcommand_ > 0 && static_cast<int>(selected_subcommands.size()) != require_subcommand_)
961  throw RequiredError(std::to_string(require_subcommand_) + " subcommand(s) required");
962 
963  // Convert missing (pairs) to extras (string only)
964  if(parent_ == nullptr) {
965  args.resize(missing()->size());
966  std::transform(std::begin(*missing()),
967  std::end(*missing()),
968  std::begin(args),
969  [](const std::pair<detail::Classifer, std::string> &val) { return val.second; });
970  std::reverse(std::begin(args), std::end(args));
971 
972  size_t num_left_over = std::count_if(
973  std::begin(*missing()), std::end(*missing()), [](std::pair<detail::Classifer, std::string> &val) {
974  return val.first != detail::Classifer::POSITIONAL_MARK;
975  });
976 
977  if(num_left_over > 0 && !(allow_extras_ || prefix_command_))
978  throw ExtrasError("[" + detail::rjoin(args, " ") + "]");
979  }
980  }

986  bool _parse_ini(std::vector<detail::ini_ret_t> &args) {
987  detail::ini_ret_t &current = args.back();
988  std::string parent = current.parent(); // respects curent.level
989  std::string name = current.name();
990 
991  // If a parent is listed, go to a subcommand
992  if(parent != "") {
993  current.level++;
994  for(const App_p &com : subcommands_)
995  if(com->check_name(parent))
996  return com->_parse_ini(args);
997  return false;
998  }
999 
1000  auto op_ptr = std::find_if(
1001  std::begin(options_), std::end(options_), [name](const Option_p &v) { return v->check_lname(name); });
1002 
1003  if(op_ptr == std::end(options_))
1004  return false;
1005 
1006  // Let's not go crazy with pointer syntax
1007  Option_p &op = *op_ptr;
1008 
1009  if(op->results_.empty()) {
1010  // Flag parsing
1011  if(op->get_expected() == 0) {
1012  if(current.inputs.size() == 1) {
1013  std::string val = current.inputs.at(0);
1014  val = detail::to_lower(val);
1015  if(val == "true" || val == "on" || val == "yes")
1016  op->results_ = {""};
1017  else if(val == "false" || val == "off" || val == "no")
1018  ;
1019  else
1020  try {
1021  size_t ui = std::stoul(val);
1022  for(size_t i = 0; i < ui; i++)
1023  op->results_.emplace_back("");
1024  } catch(const std::invalid_argument &) {
1025  throw ConversionError(current.fullname + ": Should be true/false or a number");
1026  }
1027  } else
1028  throw ConversionError(current.fullname + ": too many inputs for a flag");
1029  } else {
1030  op->results_ = current.inputs;
1031  op->run_callback();
1032  }
1033  }
1034 
1035  args.pop_back();
1036  return true;
1037  }

1041  void _parse_single(std::vector<std::string> &args, bool &positional_only) {
1042 
1043  detail::Classifer classifer = positional_only ? detail::Classifer::NONE : _recognize(args.back());
1044  switch(classifer) {
1045  case detail::Classifer::POSITIONAL_MARK:
1046  missing()->emplace_back(classifer, args.back());
1047  args.pop_back();
1048  positional_only = true;
1049  break;
1050  case detail::Classifer::SUBCOMMAND:
1051  _parse_subcommand(args);
1052  break;
1053  case detail::Classifer::LONG:
1054  // If already parsed a subcommand, don't accept options_
1055  _parse_long(args);
1056  break;
1057  case detail::Classifer::SHORT:
1058  // If already parsed a subcommand, don't accept options_
1059  _parse_short(args);
1060  break;
1061  case detail::Classifer::NONE:
1062  // Probably a positional or something for a parent (sub)command
1063  _parse_positional(args);
1064  }
1065  }

1117  void _parse_short(std::vector<std::string> &args) {
1118  std::string current = args.back();
1119 
1120  std::string name;
1121  std::string rest;
1122  if(!detail::split_short(current, name, rest))
1123  throw HorribleError("Short");
1124 
1125  auto op_ptr = std::find_if(
1126  std::begin(options_), std::end(options_), [name](const Option_p &opt) { return opt->check_sname(name); });
1127 
1128  // Option not found
1129  if(op_ptr == std::end(options_)) {
1130  // If a subcommand, try the master command
1131  if(parent_ != nullptr && fallthrough_)
1132  return parent_->_parse_short(args);
1133  // Otherwise, add to missing
1134  else {
1135  args.pop_back();
1136  missing()->emplace_back(detail::Classifer::SHORT, current);
1137  return;
1138  }
1139  }
1140 
1141  args.pop_back();
1142 
1143  // Get a reference to the pointer to make syntax bearable
1144  Option_p &op = *op_ptr;
1145 
1146  int num = op->get_expected();
1147 
1148  if(num == 0) {
1149  op->add_result("");
1150  parse_order_.push_back(op.get());
1151  } else if(rest != "") {
1152  if(num > 0)
1153  num--;
1154  op->add_result(rest);
1155  parse_order_.push_back(op.get());
1156  rest = "";
1157  }
1158 
1159  if(num == -1) {
1160  while(!args.empty() && _recognize(args.back()) == detail::Classifer::NONE) {
1161  op->add_result(args.back());
1162  parse_order_.push_back(op.get());
1163  args.pop_back();
1164  }
1165  } else
1166  while(num > 0 && !args.empty()) {
1167  num--;
1168  std::string current_ = args.back();
1169  args.pop_back();
1170  op->add_result(current_);
1171  parse_order_.push_back(op.get());
1172  }
1173 
1174  if(rest != "") {
1175  rest = "-" + rest;
1176  args.push_back(rest);
1177  }
1178  }

1181  void _parse_long(std::vector<std::string> &args) {
1182  std::string current = args.back();
1183 
1184  std::string name;
1185  std::string value;
1186  if(!detail::split_long(current, name, value))
1187  throw HorribleError("Long:" + args.back());
1188 
1189  auto op_ptr = std::find_if(
1190  std::begin(options_), std::end(options_), [name](const Option_p &v) { return v->check_lname(name); });
1191 
1192  // Option not found
1193  if(op_ptr == std::end(options_)) {
1194  // If a subcommand, try the master command
1195  if(parent_ != nullptr && fallthrough_)
1196  return parent_->_parse_long(args);
1197  // Otherwise, add to missing
1198  else {
1199  args.pop_back();
1200  missing()->emplace_back(detail::Classifer::LONG, current);
1201  return;
1202  }
1203  }
1204 
1205  args.pop_back();
1206 
1207  // Get a reference to the pointer to make syntax bearable
1208  Option_p &op = *op_ptr;
1209 
1210  int num = op->get_expected();
1211 
1212  if(value != "") {
1213  if(num != -1)
1214  num--;
1215  op->add_result(value);
1216  parse_order_.push_back(op.get());
1217  } else if(num == 0) {
1218  op->add_result("");
1219  parse_order_.push_back(op.get());
1220  }
1221 
1222  if(num == -1) {
1223  while(!args.empty() && _recognize(args.back()) == detail::Classifer::NONE) {
1224  op->add_result(args.back());
1225  parse_order_.push_back(op.get());
1226  args.pop_back();
1227  }
1228  } else
1229  while(num > 0 && !args.empty()) {
1230  num--;
1231  op->add_result(args.back());
1232  parse_order_.push_back(op.get());
1233  args.pop_back();
1234  }
1235  return;
1236  }
