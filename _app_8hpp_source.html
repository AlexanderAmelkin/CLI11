803  std::string config_to_str(bool default_also = false, std::string prefix = "") const {
804  std::stringstream out;
805  for(const Option_p &opt : options_) {
806 
807  // Only process option with a long-name
808  if(!opt->lnames_.empty()) {
809  std::string name = prefix + opt->lnames_[0];
810 
811  // Non-flags
812  if(opt->get_expected() != 0) {
813 
814  // If the option was found on command line
815  if(opt->count() > 0)
816  out << name << "=" << detail::inijoin(opt->results()) << std::endl;
817 
818  // If the option has a default and is requested by optional argument
819  else if(default_also && !opt->defaultval_.empty())
820  out << name << "=" << opt->defaultval_ << std::endl;
821  // Flag, one passed
822  } else if(opt->count() == 1) {
823  out << name << "=true" << std::endl;
824 
825  // Flag, multiple passed
826  } else if(opt->count() > 1) {
827  out << name << "=" << opt->count() << std::endl;
828 
829  // Flag, not present
830  } else if(opt->count() == 0 && default_also && opt.get() != get_help_ptr()) {
831  out << name << "=false" << std::endl;
832  }
833  }
834  }
835  for(const App_p &subcom : subcommands_)
836  out << subcom->config_to_str(default_also, prefix + subcom->name_ + ".");
837  return out.str();
838  }

841  std::string help(size_t wid = 30, std::string prev = "") const {
842  // Delegate to subcommand if needed
843  if(prev.empty())
844  prev = name_;
845  else
846  prev += " " + name_;
847 
848  auto selected_subcommands = get_subcommands();
849  if(!selected_subcommands.empty())
850  return selected_subcommands.at(0)->help(wid, prev);
851 
852  std::stringstream out;
853  out << description_ << std::endl;
854  out << "Usage: " << prev;
855 
856  // Check for options_
857  bool npos = false;
858  std::set<std::string> groups;
859  for(const Option_p &opt : options_) {
860  if(opt->nonpositional()) {
861  npos = true;
862  groups.insert(opt->get_group());
863  }
864  }
865 
866  if(npos)
867  out << " [OPTIONS]";
868

1189  bool _parse_ini(std::vector<detail::ini_ret_t> &args) {
1190  detail::ini_ret_t &current = args.back();
1191  std::string parent = current.parent(); // respects current.level
1192  std::string name = current.name();
1193 
1194  // If a parent is listed, go to a subcommand
1195  if(!parent.empty()) {
1196  current.level++;

1198  if(com->check_name(parent))
1199  return com->_parse_ini(args);
1200  return false;
1201  }
1202

1204  std::begin(options_), std::end(options_), [name](const Option_p &v) { return v->check_lname(name); });
1205 
1206  if(op_ptr == std::end(options_))
1207  return false;
1208 
1209  // Let's not go crazy with pointer syntax
1210  Option_p &op = *op_ptr;
1211 
1212  if(op->results_.empty()) {
1213  // Flag parsing
1214  if(op->get_expected() == 0) {

1217  val = detail::to_lower(val);
1218  if(val == "true" || val == "on" || val == "yes")
1219  op->results_ = {""};
1220  else if(val == "false" || val == "off" || val == "no")
1221  ;
1222  else
1223  try {
1224  size_t ui = std::stoul(val);
1225  for(size_t i = 0; i < ui; i++)

1228  throw ConversionError(current.fullname + ": Should be true/false or a number");
1229  }
1230  } else
1231  throw ConversionError(current.fullname + ": too many inputs for a flag");
1232  } else {
1233  op->results_ = current.inputs;
1234  op->run_callback();
1235  }
1236  }
1237 
1238  args.pop_back();
1239  return true;

1335  void _parse_arg(std::vector<std::string> &args, bool second_dash) {
1336 
1337  detail::Classifer current_type = second_dash ? detail::Classifer::LONG : detail::Classifer::SHORT;
1338 
1339  std::string current = args.back();
1340 
1341  std::string name;
1342  std::string value;
1343  std::string rest;
1344 
1345  if(second_dash) {
1346  if(!detail::split_long(current, name, value))
1347  throw HorribleError("Long parsed but missing (you should not see this):" + args.back());
1348  } else {
1349  if(!detail::split_short(current, name, rest))
1350  throw HorribleError("Short parsed but missing! You should not see this");
1351  }
1352 
1353  auto op_ptr = std::find_if(std::begin(options_), std::end(options_), [name, second_dash](const Option_p &opt) {
1354  return second_dash ? opt->check_lname(name) : opt->check_sname(name);
1355  });
1356 
1357  // Option not found
1358  if(op_ptr == std::end(options_)) {
1359  // If a subcommand, try the master command
1360  if(parent_ != nullptr && fallthrough_)
1361  return parent_->_parse_arg(args, second_dash);
1362  // Otherwise, add to missing
1363  else {
1364  args.pop_back();
1365  missing_.emplace_back(current_type, current);
1366  return;
1367  }
1368  }
1369 
1370  args.pop_back();
1371 
1372  // Get a reference to the pointer to make syntax bearable
1373  Option_p &op = *op_ptr;
1374 
1375  int num = op->get_expected();
1376 
1377  if(!value.empty()) {
1378  if(num != -1)
1379  num--;
1380  op->add_result(value);
1381  parse_order_.push_back(op.get());
1382  } else if(num == 0) {
1383  op->add_result("");
1384  parse_order_.push_back(op.get());
1385  } else if(!rest.empty()) {
1386  if(num > 0)
1387  num--;
1388  op->add_result(rest);
1389  parse_order_.push_back(op.get());
1390  rest = "";
1391  }
1392 
1393  // Unlimited vector parser
1394  if(num < 0) {
1395  int collected = 0; // Make sure we always eat the minimum
1396  while(!args.empty() && _recognize(args.back()) == detail::Classifer::NONE) {
1397  if(collected >= -num) {
1398  // We could break here for allow extras, but we don't
1399 
1400  // If any positionals remain, don't keep eating
1401  if(_count_remaining_positionals() > 0)
1402  break;
1403 
1404  // If there are any unlimited positionals, those also take priority
1405  if(std::any_of(std::begin(options_), std::end(options_), [](const Option_p &opt) {
1406  return opt->get_positional() && opt->get_expected() < 0;
1407  }))
1408  break;
1409  }
1410  op->add_result(args.back());
1411  parse_order_.push_back(op.get());
1412  args.pop_back();
1413  collected++;
1414  }
1415 
1416  } else {
1417  while(num > 0 && !args.empty()) {
1418  num--;
1419  std::string current_ = args.back();
1420  args.pop_back();
1421  op->add_result(current_);
1422  parse_order_.push_back(op.get());
1423  }
1424 
1425  if(num > 0) {
1426  throw ArgumentMismatch(op->single_name() + ": " + std::to_string(num) + " required " +
1427  op->get_type_name() + " missing");
1428  }
1429  }
1430 
1431  if(!rest.empty()) {
1432  rest = "-" + rest;
1433  args.push_back(rest);
1434  }
1435  }
