557  std::string help(size_t wid=30, std::string prev="") const {
558  // Delegate to subcommand if needed
559  if(prev == "")
560  prev = name_;
561  else
562  prev += " " + name_;
563 
564  if(selected_subcommands_.size() > 0)
565  return selected_subcommands_.at(0)->help(wid, prev);
566 
567  std::stringstream out;
568  out << description_ << std::endl;
569  out << "Usage: " << prev;
570 
571  // Check for options_
572  bool npos = false;
573  std::set<std::string> groups;
574  for(const Option_p &opt : options_) {
575  if(opt->nonpositional()) {
576  npos = true;
577  groups.insert(opt->get_group());
578  }
579  }
580 
581  if(npos)
582  out << " [OPTIONS]";
583 
584  // Positionals
585  bool pos=false;
586  for(const Option_p &opt : options_)
587  if(opt->get_positional()) {
588  // A hidden positional should still show up in the usage statement
589  //if(detail::to_lower(opt->get_group()) == "hidden")
590  // continue;
591  out << " " << opt->help_positional();
592  if(opt->has_description())
593  pos=true;
594  }
595 
596  if(subcommands_.size() > 0) {
597  if(require_subcommand_ != 0)
598  out << " SUBCOMMAND";
599  else
600  out << " [SUBCOMMAND]";
601  }
602 
603  out << std::endl << std::endl;
604 
605  // Positional descriptions
606  if(pos) {
607  out << "Positionals:" << std::endl;
608  for(const Option_p &opt : options_) {
609  if(detail::to_lower(opt->get_group()) == "hidden")
610  continue;
611  if(opt->get_positional() && opt->has_description())
612  detail::format_help(out, opt->help_pname(), opt->get_description(), wid);
613  }
614  out << std::endl;
615 
616  }
617 
618 
619  // Options
620  if(npos) {
621  for (const std::string& group : groups) {
622  if(detail::to_lower(group) == "hidden")
623  continue;
624  out << group << ":" << std::endl;
625  for(const Option_p &opt : options_) {
626  if(opt->nonpositional() && opt->get_group() == group)
627  detail::format_help(out, opt->help_name(), opt->get_description(), wid);
628 
629  }
630  out << std::endl;
631  }
632  }
633 
634  // Subcommands
635  if(subcommands_.size()> 0) {
636  out << "Subcommands:" << std::endl;
637  for(const App_p &com : subcommands_)
638  detail::format_help(out, com->get_name(), com->description_, wid);
639  }
640  return out.str();
641  }

708  detail::Classifer _recognize(const std::string &current) const {
709  std::string dummy1, dummy2;
710 
711  if(current == "--")
712  return detail::Classifer::POSITIONAL_MARK;
713  if(_valid_subcommand(current))
714  return detail::Classifer::SUBCOMMAND;
715  if(detail::split_long(current, dummy1, dummy2))

724  void _parse(std::vector<std::string> &args) {
725  bool positional_only = false;
726 
727  while(args.size()>0) {
728  _parse_single(args, positional_only);
729  }
730 
731  if (help_ptr_ != nullptr && help_ptr_->count() > 0) {
732  throw CallForHelp();
733  }
734 
735 
736  // Process an INI file
737  if (config_ptr_ != nullptr && config_name_ != "") {
738  try {
739  std::vector<detail::ini_ret_t> values = detail::parse_ini(config_name_);
740  while(values.size() > 0) {
741  if(!_parse_env(values)) {
742  throw ExtrasError(values.back().name());
743  }
744  }
745  } catch (const FileError &) {
746  if(config_required_)
747  throw;
748  }
749  }
750 
751 
752  // Get envname options if not yet passed
753  for(const Option_p& opt : options_) {
754  if (opt->count() == 0 && opt->envname_ != "") {
755  char *ename = std::getenv(opt->envname_.c_str());
756  if(ename != nullptr) {
757  opt->add_result(std::string(ename));
758  }
759  }
760  }
761 
762  // Process callbacks
763  for(const Option_p& opt : options_) {
764  if (opt->count() > 0) {
765  opt->run_callback();
766  }
767  }
768 
769  // Verify required options
770  for(const Option_p& opt : options_) {
771  // Required
772  if (opt->get_required()
773  && (opt->count() < opt->get_expected() || opt->count() == 0))
774  throw RequiredError(opt->get_name());
775  // Requires
776  for (const Option* opt_req : opt->requires_)
777  if (opt->count() > 0 && opt_req->count() == 0)
778  throw RequiresError(opt->get_name(), opt_req->get_name());
779  // Excludes
780  for (const Option* opt_ex : opt->excludes_)
781  if (opt->count() > 0 && opt_ex->count() != 0)
782  throw ExcludesError(opt->get_name(), opt_ex->get_name());
783  }
784 
785  if(require_subcommand_ < 0 && selected_subcommands_.size() == 0)
786  throw RequiredError("Subcommand required");
787  else if(require_subcommand_ > 0 && (int) selected_subcommands_.size() != require_subcommand_)
788  throw RequiredError(std::to_string(require_subcommand_) + " subcommand(s) required");
789 
790  // Convert missing (pairs) to extras (string only)
791  if(parent_ == nullptr) {
792  args.resize(missing()->size());
793  std::transform(std::begin(*missing()), std::end(*missing()), std::begin(args),
794  [](const std::pair<detail::Classifer, std::string>& val){return val.second;});
795  std::reverse(std::begin(args), std::end(args));
796 
797  size_t num_left_over = std::count_if(std::begin(*missing()), std::end(*missing()),
798  [](std::pair<detail::Classifer, std::string>& val){return val.first != detail::Classifer::POSITIONAL_MARK;});
799 
800  if(num_left_over>0 && !allow_extras_)
801  throw ExtrasError("[" + detail::rjoin(args, " ") + "]");
802  }
803  }

809  bool _parse_env(std::vector<detail::ini_ret_t> &args) {
810  detail::ini_ret_t& current = args.back();
811  std::string parent = current.parent();
812  std::string name = current.name();
813  if(parent != "") {
814  current.level++;
815  for(const App_p &com : subcommands_)
816  if(com->check_name(parent))
817  return com->_parse_env(args);
818  return false;
819  }
820 
821  auto op_ptr = std::find_if(std::begin(options_), std::end(options_),
822  [name](const Option_p &v){return v->check_lname(name);});
823 
824  if(op_ptr == std::end(options_))
825  return false;
826 
827  // Let's not go crazy with pointer syntax
828  Option_p& op = *op_ptr;
829 
830 
831  if(op->results_.empty()) {
832  // Flag parsing
833  if(op->get_expected() == 0) {
834  if(current.inputs.size() == 1) {
835  std::string val = current.inputs.at(0);
836  val = detail::to_lower(val);
837  if(val == "true" || val == "on" || val == "yes")
838  op->results_ = {""};
839  else if(val == "false" || val == "off" || val == "no")
840  ;
841  else
842  try {
843  size_t ui = std::stoul(val);
844  for (size_t i=0; i<ui; i++)
845  op->results_.push_back("");
846  } catch (const std::invalid_argument &) {
847  throw ConversionError(current.fullname + ": Should be true/false or a number");
848  }
849  } else
850  throw ConversionError(current.fullname + ": too many inputs for a flag");
851  } else
852  op->results_ = current.inputs;
853  }
854 
855  args.pop_back();
856  return true;
857  }

860  void _parse_single(std::vector<std::string> &args, bool &positional_only) {
861 
862  detail::Classifer classifer = positional_only ? detail::Classifer::NONE : _recognize(args.back());
863  switch(classifer) {
864  case detail::Classifer::POSITIONAL_MARK:
865  missing()->emplace_back(classifer, args.back());
866  args.pop_back();
867  positional_only = true;
868  break;
869  case detail::Classifer::SUBCOMMAND:
870  _parse_subcommand(args);
871  break;
872  case detail::Classifer::LONG:
873  // If already parsed a subcommand, don't accept options_
874  _parse_long(args);
875  break;
876  case detail::Classifer::SHORT:
877  // If already parsed a subcommand, don't accept options_
878  _parse_short(args);
879  break;
880  case detail::Classifer::NONE:
881  // Probably a positional or something for a parent (sub)command
882  _parse_positional(args);
883  }

889  void _parse_positional(std::vector<std::string> &args) {
890 
891  std::string positional = args.back();
892  for(const Option_p& opt : options_) {
