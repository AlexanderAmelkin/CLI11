433  void run_callback() {
434 
435  // Run the validators (can change the string)
436  if(!validators_.empty()) {
437  for(std::string &result : results_)
438  for(const std::function<std::string(std::string &)> &vali : validators_) {
439  std::string err_msg = vali(result);
440  if(!err_msg.empty())
441  throw ValidationError(single_name(), err_msg);
442  }
443  }
444 
445  bool local_result;
446  // If take_last, only operate on the final item
447  if(last_) {
448  results_t partial_result = {results_.back()};
449  local_result = !callback_(partial_result);
450  } else {
451  if((expected_ > 0 && results_.size() != static_cast<size_t>(expected_)) ||
452  (expected_ < 0 && results_.size() < static_cast<size_t>(-expected_)))
453  throw ArgumentMismatch(single_name(), expected_, results_.size());
454  else
455  local_result = !callback_(results_);
456  }
457 
458  if(local_result)
459  throw ConversionError("Could not convert: " + get_name() + "=" + detail::join(results_));
460  }

463  bool operator==(const Option &other) const {
464  for(const std::string &sname : snames_)
465  if(other.check_sname(sname))
466  return true;
467  for(const std::string &lname : lnames_)
468  if(other.check_lname(lname))
469  return true;
470  // We need to do the inverse, just in case we are ignore_case
471  for(const std::string &sname : other.snames_)
472  if(check_sname(sname))
473  return true;
474  for(const std::string &lname : other.lnames_)
475  if(check_lname(lname))
476  return true;
477  return false;
478  }

481  bool check_name(std::string name) const {
482 
483  if(name.length() > 2 && name.substr(0, 2) == "--")
484  return check_lname(name.substr(2));
485  else if(name.length() > 1 && name.substr(0, 1) == "-")
486  return check_sname(name.substr(1));
487  else {
488  std::string local_pname = pname_;
489  if(ignore_case_) {
490  local_pname = detail::to_lower(local_pname);
491  name = detail::to_lower(name);
492  }
493  return name == local_pname;
494  }
495  }

498  bool check_sname(std::string name) const {
499  if(ignore_case_) {
500  name = detail::to_lower(name);
501  return std::find_if(std::begin(snames_), std::end(snames_), [&name](std::string local_sname) {
502  return detail::to_lower(local_sname) == name;
503  }) != std::end(snames_);
504  } else
505  return std::find(std::begin(snames_), std::end(snames_), name) != std::end(snames_);
506  }
