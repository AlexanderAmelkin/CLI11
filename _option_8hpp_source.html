376  void run_callback() const {
377  if(!callback_(results_))
378  throw ConversionError(get_name() + "=" + detail::join(results_));
379  if(validators_.size()>0) {

388  bool operator== (const Option& other) const {
389  for(const std::string &sname : snames_)

392  for(const std::string &lname : lnames_)
393  if(other.check_lname(lname))

397  if(check_sname(sname))
398  return true;
399  for(const std::string &lname : other.lnames_)
400  if(check_lname(lname))
401  return true;
402  return false;

406  bool check_name(std::string name) const {
407 
408  if(name.length()>2 && name.substr(0,2) == "--")
409  return check_lname(name.substr(2));
410  else if (name.length()>1 && name.substr(0,1) == "-")
411  return check_sname(name.substr(1));
412  else {
413  std::string local_pname = pname_;
414  if(ignore_case_) {
415  local_pname = detail::to_lower(local_pname);
416  name = detail::to_lower(name);
417  }
418  return name == local_pname;
419  }
420  }

423  bool check_sname(std::string name) const {
424  if(ignore_case_) {
425  name = detail::to_lower(name);
426  return std::find_if(std::begin(snames_), std::end(snames_),
427  [&name](std::string local_sname){return detail::to_lower(local_sname) == name;})
428  != std::end(snames_);
429  } else
430  return std::find(std::begin(snames_), std::end(snames_), name) != std::end(snames_);
431  }
