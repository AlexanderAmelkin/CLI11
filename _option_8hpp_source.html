420  void run_callback() const {
421  bool local_result;
422  // If take_last, only operate on the final item
423  if(last_) {
424  results_t partial_result = {results_.back()};
425  local_result = !callback_(partial_result);
426  } else {
427  local_result = !callback_(results_);
428  }
429 
430  if(local_result)
431  throw ConversionError(get_name() + "=" + detail::join(results_));
432 
433  if(!validators_.empty()) {
434  for(const std::string &result : results_)
435  for(const std::function<bool(std::string)> &vali : validators_)
436  if(!vali(result))
437  throw ValidationError(get_name() + "=" + result);
438  }
439  }

460  bool check_name(std::string name) const {
461 
462  if(name.length() > 2 && name.substr(0, 2) == "--")
463  return check_lname(name.substr(2));
464  else if(name.length() > 1 && name.substr(0, 1) == "-")
465  return check_sname(name.substr(1));
466  else {
467  std::string local_pname = pname_;
468  if(ignore_case_) {
469  local_pname = detail::to_lower(local_pname);
470  name = detail::to_lower(name);
471  }
472  return name == local_pname;
473  }
474  }

477  bool check_sname(std::string name) const {
478  if(ignore_case_) {
479  name = detail::to_lower(name);
480  return std::find_if(std::begin(snames_), std::end(snames_), [&name](std::string local_sname) {
481  return detail::to_lower(local_sname) == name;
482  }) != std::end(snames_);
483  } else
484  return std::find(std::begin(snames_), std::end(snames_), name) != std::end(snames_);
485  }
