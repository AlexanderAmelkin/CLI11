487  std::string get_name(bool positional = false, //<[input] Show the positional name
488  bool all_options = false //<[input] Show every option
489  ) const {
490 
491  if(all_options) {
492 
493  std::vector<std::string> name_list;
494 
496  if((positional && pname_.length()) || (snames_.empty() && lnames_.empty()))
497  name_list.push_back(pname_);
498 
499  for(const std::string &sname : snames_)
500  name_list.push_back("-" + sname);
501 
502  for(const std::string &lname : lnames_)
503  name_list.push_back("--" + lname);
504 
505  return detail::join(name_list);
506 
507  } else {
508 
509  // This returns the positional name no matter what
510  if(positional)
511  return pname_;
512 
513  // Prefer long name
514  else if(!lnames_.empty())
515  return std::string("--") + lnames_[0];
516 
517  // Or short name if no long name
518  else if(!snames_.empty())
519  return std::string("-") + snames_[0];
520 
521  // If positional is the only name, it's okay to use that
522  else
523  return pname_;
524  }
525  }

532  void run_callback() {
533 
534  // Run the validators (can change the string)
535  if(!validators_.empty()) {
536  for(std::string &result : results_)
537  for(const std::function<std::string(std::string &)> &vali : validators_) {
538  std::string err_msg = vali(result);
539  if(!err_msg.empty())
540  throw ValidationError(get_name(), err_msg);
541  }
542  }
543 
544  bool local_result;
545 
546  // Num items expected or length of vector, always at least 1
547  // Only valid for a trimming policy
548  int trim_size = std::min(std::max(std::abs(get_items_expected()), 1), static_cast<int>(results_.size()));
549 
550  // Operation depends on the policy setting
551  if(multi_option_policy_ == MultiOptionPolicy::TakeLast) {
552  // Allow multi-option sizes (including 0)
553  results_t partial_result{results_.end() - trim_size, results_.end()};
554  local_result = !callback_(partial_result);
555 
556  } else if(multi_option_policy_ == MultiOptionPolicy::TakeFirst) {
557  results_t partial_result{results_.begin(), results_.begin() + trim_size};
558  local_result = !callback_(partial_result);
559 
560  } else if(multi_option_policy_ == MultiOptionPolicy::Join) {
561  results_t partial_result = {detail::join(results_, "\n")};
562  local_result = !callback_(partial_result);
563 
564  } else {
565  // Exact number required
566  if(get_items_expected() > 0) {
567  if(results_.size() != static_cast<size_t>(get_items_expected()))
568  throw ArgumentMismatch(get_name(), get_items_expected(), results_.size());
569  // Variable length list
570  } else if(get_items_expected() < 0) {
571  // Require that this be a multiple of expected size and at least as many as expected
572  if(results_.size() < static_cast<size_t>(-get_items_expected()) ||
573  results_.size() % static_cast<size_t>(std::abs(get_type_size())) != 0)
574  throw ArgumentMismatch(get_name(), get_items_expected(), results_.size());
575  }
576  local_result = !callback_(results_);
577  }
578 
579  if(local_result)
580  throw ConversionError(get_name(), results_);
581  }

584  bool operator==(const Option &other) const {
585  for(const std::string &sname : snames_)
586  if(other.check_sname(sname))
587  return true;
588  for(const std::string &lname : lnames_)

592  for(const std::string &sname : other.snames_)
593  if(check_sname(sname))
594  return true;
595  for(const std::string &lname : other.lnames_)
596  if(check_lname(lname))
597  return true;
598  return false;
599  }

602  bool check_name(std::string name) const {
603 
604  if(name.length() > 2 && name.substr(0, 2) == "--")
605  return check_lname(name.substr(2));
606  else if(name.length() > 1 && name.substr(0, 1) == "-")
607  return check_sname(name.substr(1));
608  else {
609  std::string local_pname = pname_;
610  if(ignore_case_) {
611  local_pname = detail::to_lower(local_pname);
612  name = detail::to_lower(name);
613  }
614  return name == local_pname;
615  }
616  }

619  bool check_sname(std::string name) const {
620  if(ignore_case_) {
621  name = detail::to_lower(name);
622  return std::find_if(std::begin(snames_), std::end(snames_), [&name](std::string local_sname) {
623  return detail::to_lower(local_sname) == name;
624  }) != std::end(snames_);
625  } else
626  return std::find(std::begin(snames_), std::end(snames_), name) != std::end(snames_);
627  }
