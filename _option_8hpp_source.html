462  void run_callback() {
463 
464  // Run the validators (can change the string)
465  if(!validators_.empty()) {
466  for(std::string &result : results_)
467  for(const std::function<std::string(std::string &)> &vali : validators_) {
468  std::string err_msg = vali(result);
469  if(!err_msg.empty())
470  throw ValidationError(single_name(), err_msg);
471  }
472  }
473 
474  bool local_result;
475 
476  // Operation depends on the policy setting
477  if(multi_option_policy_ == MultiOptionPolicy::TakeLast) {
478  results_t partial_result = {results_.back()};
479  local_result = !callback_(partial_result);
480  } else if(multi_option_policy_ == MultiOptionPolicy::TakeFirst) {
481  results_t partial_result = {results_.at(0)};
482  local_result = !callback_(partial_result);
483  } else if(multi_option_policy_ == MultiOptionPolicy::Join) {
484  results_t partial_result = {detail::join(results_, "\n")};
485  local_result = !callback_(partial_result);
486  } else {
487  if((expected_ > 0 && results_.size() != static_cast<size_t>(expected_)) ||
488  (expected_ < 0 && results_.size() < static_cast<size_t>(-expected_)))
489  throw ArgumentMismatch(single_name(), expected_, results_.size());
490  else
491  local_result = !callback_(results_);
492  }
493 
494  if(local_result)
495  throw ConversionError(get_name(), results_);
496  }

499  bool operator==(const Option &other) const {
500  for(const std::string &sname : snames_)
501  if(other.check_sname(sname))
502  return true;
503  for(const std::string &lname : lnames_)

507  for(const std::string &sname : other.snames_)
508  if(check_sname(sname))
509  return true;
510  for(const std::string &lname : other.lnames_)
511  if(check_lname(lname))
512  return true;
513  return false;
514  }

517  bool check_name(std::string name) const {
518 
519  if(name.length() > 2 && name.substr(0, 2) == "--")
520  return check_lname(name.substr(2));
521  else if(name.length() > 1 && name.substr(0, 1) == "-")
522  return check_sname(name.substr(1));
523  else {
524  std::string local_pname = pname_;
525  if(ignore_case_) {
526  local_pname = detail::to_lower(local_pname);
527  name = detail::to_lower(name);
528  }
529  return name == local_pname;
530  }
531  }
