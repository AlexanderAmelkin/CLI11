358  void run_callback() const {
359  bool result;
360  // If take_last, only operate on the final item
361  if(last_) {
362  results_t partial_result = {results_.back()};
363  result = !callback_(partial_result);
364  } else {
365  result = !callback_(results_);
366  }
367 
368  if(result)
369  throw ConversionError(get_name() + "=" + detail::join(results_));
370  if(!validators_.empty()) {
371  for(const std::string &result : results_)
372  for(const std::function<bool(std::string)> &vali : validators_)
373  if(!vali(result))
374  throw ValidationError(get_name() + "=" + result);
375  }
376  }

379  bool operator==(const Option &other) const {
380  for(const std::string &sname : snames_)
381  if(other.check_sname(sname))
382  return true;
383  for(const std::string &lname : lnames_)
384  if(other.check_lname(lname))

386  // We need to do the inverse, just in case we are ignore_case
387  for(const std::string &sname : other.snames_)
388  if(check_sname(sname))
389  return true;
390  for(const std::string &lname : other.lnames_)
391  if(check_lname(lname))
392  return true;
393  return false;

397  bool check_name(std::string name) const {
398 
399  if(name.length() > 2 && name.substr(0, 2) == "--")
400  return check_lname(name.substr(2));
401  else if(name.length() > 1 && name.substr(0, 1) == "-")
402  return check_sname(name.substr(1));
403  else {
404  std::string local_pname = pname_;
405  if(ignore_case_) {
406  local_pname = detail::to_lower(local_pname);
407  name = detail::to_lower(name);
408  }
409  return name == local_pname;
410  }
411  }

414  bool check_sname(std::string name) const {
415  if(ignore_case_) {
416  name = detail::to_lower(name);
417  return std::find_if(std::begin(snames_), std::end(snames_), [&name](std::string local_sname) {
418  return detail::to_lower(local_sname) == name;
419  }) != std::end(snames_);
420  } else
421  return std::find(std::begin(snames_), std::end(snames_), name) != std::end(snames_);
422  }
