262  bool check_name(std::string name) const {
263 
264  if(name.length()>2 && name.substr(0,2) == "--")
265  return check_lname(name.substr(2));
266  else if (name.length()>1 && name.substr(0,1) == "-")
267  return check_sname(name.substr(1));
268  else {
269  std::string local_pname = pname;
270  if(case_insensitive) {
271  local_pname = detail::to_lower(local_pname);
272  name = detail::to_lower(name);
273  }
274  return name == local_pname;
275  }
276  }

279  bool check_sname(std::string name) const {
280  if(case_insensitive) {
281  name = detail::to_lower(name);
282  return std::find_if(std::begin(snames), std::end(snames),
283  [&name](std::string local_sname){return detail::to_lower(local_sname) == name;})
284  != std::end(snames);
285  } else
286  return std::find(std::begin(snames), std::end(snames), name) != std::end(snames);
287  }
