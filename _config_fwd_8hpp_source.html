20 inline std::string ini_join(std::vector<std::string> args) {
21  std::ostringstream s;
22  size_t start = 0;
23  for(const auto &arg : args) {
24  if(start++ > 0)
25  s << " ";
26 
27  auto it = std::find_if(arg.begin(), arg.end(), [](char ch) { return std::isspace<char>(ch, std::locale()); });
28  if(it == arg.end())
29  s << arg;
30  else if(arg.find(R"(")") == std::string::npos)
31  s << R"(")" << arg << R"(")";
32  else
33  s << R"(')" << arg << R"(')";
34  }
35 
36  return s.str();
37 }

102  std::vector<ConfigItem> from_config(std::istream &input) const override {
103  std::string line;
104  std::string section = "default";
105 
106  std::vector<ConfigItem> output;
107 
108  while(getline(input, line)) {
109  std::vector<std::string> items_buffer;
110 
111  detail::trim(line);
112  size_t len = line.length();
113  if(len > 1 && line[0] == '[' && line[len - 1] == ']') {
114  section = line.substr(1, len - 2);
115  } else if(len > 0 && line[0] != ';') {
116  output.emplace_back();
117  ConfigItem &out = output.back();
118 
119  // Find = in string, split and recombine
120  auto pos = line.find('=');
121  if(pos != std::string::npos) {
122  out.name = detail::trim_copy(line.substr(0, pos));
123  std::string item = detail::trim_copy(line.substr(pos + 1));
124  items_buffer = detail::split_up(item);
125  } else {
126  out.name = detail::trim_copy(line);
127  items_buffer = {"ON"};
128  }
129 
130  if(detail::to_lower(section) != "default") {
131  out.parents = {section};
132  }
133 
134  if(out.name.find('.') != std::string::npos) {
135  std::vector<std::string> plist = detail::split(out.name, '.');
136  out.name = plist.back();
137  plist.pop_back();
138  out.parents.insert(out.parents.end(), plist.begin(), plist.end());
139  }
140 
141  out.inputs.insert(std::end(out.inputs), std::begin(items_buffer), std::end(items_buffer));
142  }
143  }
144  return output;
145  }
