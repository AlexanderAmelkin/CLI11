60  section = line.substr(1,len-2);
61  } else if (len > 0 && line[0] != ';') {
62  output.emplace_back();

68  name = detail::trim_copy(line.substr(0,pos));
69  std::string item = detail::trim_copy(line.substr(pos+1));
70  items = detail::split_up(item);
71  } else
72  name = detail::trim_copy(line);
73

75  out.fullname = name;
76  else
77  out.fullname = section + "." + name;
78 
79  out.inputs.insert(std::end(out.inputs), std::begin(items), std::end(items));
80  }
81  }
